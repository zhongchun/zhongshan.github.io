Java 使用 protobuf
Java Protobuf

简介

Protocol buffers are Google’s language-neutral, platform-neutral, extensible mechanism for serializing structured data – think XML, but smaller, faster, and simpler. You define how you want your data to be structured once, then you can use special generated source code to easily write and read your structured data to and from a variety of data streams and using a variety of languages.

mac 安装 protobuf

brew install protobuf

安装后：

protoc --version

输出：

libprotoc 3.6.1

定义 proto 文件

person.proto

syntax = "proto3";
option java_package = "com.dragon.bermaker";
option java_outer_classname = "PersonModel";

message Person {
     int32 id = 1;
     string name = 2;
     string email = 3;
}

生成 Java 文件

执行：

protoc -I=$SRC_DIR --java_out=$DST_DIR $SRC_DIR/person.proto

测试 protobuf

maven pom 文件添加依赖：

<!-- protobuf -->
<dependency>
    <groupId>com.google.protobuf</groupId>
    <artifactId>protobuf-java</artifactId>
    <version>${protobuf.version}</version>
</dependency>

<dependency>
    <groupId>com.google.protobuf</groupId>
    <artifactId>protobuf-java-util</artifactId>
    <version>${protobuf.version}</version>
</dependency>

测试代码：

/*

package com.dragon.bermaker.bean;

import org.junit.Test;

import com.google.protobuf.InvalidProtocolBufferException;
import com.google.protobuf.util.JsonFormat;

/**
 * @ClassName: PersonModelTest
 * @Project: dragon-wing
 * @Description: TODO
 * @Author: bermaker
 * @Date: 2019/1/30 5:49 PM
 * @Version: 1.0
 */
public class PersonModelTest {

    @Test
    public void testProtobuf() {

        PersonModel.Person.Builder builder = PersonModel.Person.newBuilder();
        builder.setId(1);
        builder.setName("tiantian");
        builder.setEmail("tiantian@qq.com");

        PersonModel.Person person = builder.build();
        System.out.println("Before: ");
        System.out.println(person);

        byte[] buff = person.toByteArray();
        System.out.println("=======================================================================");
        for (byte b : buff) {
            System.out.print(b);
        }
        System.out.println();
        System.out.println("Buf length=" + buff.length);
        System.out.println("=======================================================================");


        PersonModel.Person personParsed = null;
        try {
            personParsed = PersonModel.Person.parseFrom(buff);
            System.out.println("After: ");
            System.out.println(personParsed);
        } catch (InvalidProtocolBufferException e) {
            e.printStackTrace();
        }

        // 转换成json
        String json = "";
        try {
            json = JsonFormat.printer().print(builder);
            System.out.println(json.toString());
            System.out.println("Json length=" + json.getBytes().length);
        } catch (InvalidProtocolBufferException e) {
            e.printStackTrace();
        }

    }

}
